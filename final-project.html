get_wfh_by_zip <- function(year) {
  get_acs(
    geography = "zip code tabulation area",
    variables = c(wfh = "B08006_017"),  # Worked from home
    year = year,
    survey = "acs5",                    # Change to 5-year ACS
    output = "wide"
  ) |>
    select(GEOID, wfhE) |>
    rename(zip = GEOID, !!paste0("wfh_", year) := wfhE)
}
